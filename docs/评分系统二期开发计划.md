### 5.1 语义增强的本地评分器（原2.1）

#### 核心思想

引入轻量级语义模型（Sentence-BERT），增强本地评分的语义理解能力。

#### 技术实现

```python
"""
增强的本地评分器 - 增加语义理解能力
"""
import logging
import re
from typing import List, Dict, Any, Optional
from pathlib import Path
from dataclasses import dataclass, field
import math

logger = logging.getLogger(__name__)


@dataclass
class EnhancedScoredClip:
    """增强版评分片段"""
    id: int
    start_time: str
    end_time: str
    content: str
    final_score: float
    scoring_method: str = "enhanced_local"
    quality_note: str = "[INFO] 增强版本地评分"
    features: Dict[str, float] = field(default_factory=dict)
    confidence: float = 0.0  # 新增：评分置信度


class EnhancedLocalScorer:
    """增强版本地评分器"""

    def __init__(self):
        self.config = {
            # 各维度权重
            "semantic_weight": 0.35,      # 新增：语义评分
            "length_weight": 0.20,         # 调整：字幕长度权重降低
            "energy_weight": 0.15,         # 调整：音频能量权重降低
            "diversity_weight": 0.15,      # 词汇多样性
            "keyword_weight": 0.15,        # 关键词检测

            # 语义评分配置
            "semantic_model": "all-MiniLM-L6-v2",
            "semantic_threshold": 0.5,

            # 字幕长度配置（更宽松）
            "length_optimal_min": 15,
            "length_optimal_max": 100,
            "length_good_min": 5,
            "length_good_max": 200,
        }

        # 预定义的"高质量内容"关键词向量（用于语义匹配）
        self._high_quality_topics = [
            "核心要点", "关键发现", "重要结论", "解决方案",
            "创新方法", "数据分析", "研究结果", "实战经验",
            "专业知识", "深度分析", "关键技术", "最佳实践"
        ]

        # 否定词模式（用于扣分）
        self._negative_patterns = [
            re.compile(r'(完全)?无效|错误|失败|不行|不能'),
            re.compile(r'不重要|没什么|无所谓'),
            re.compile(r'一般|普通|还行|将就')
        ]

        # 语义模型（延迟加载）
        self._semantic_model = None

    def _load_semantic_model(self):
        """延迟加载语义模型"""
        if self._semantic_model is None:
            try:
                from sentence_transformers import SentenceTransformer, util
                self._semantic_model = SentenceTransformer(self.config["semantic_model"])
                self._semantic_util = util
                logger.info("成功加载语义模型")
            except ImportError:
                logger.warning("未安装sentence_transformers，语义评分将被跳过")

    def _score_semantic_relevance(self, content: str) -> float:
        """语义相关性评分 - 判断内容与高质量话题的相似度"""
        if not content:
            return 0.3

        self._load_semantic_model()
        if self._semantic_model is None:
            return 0.5  # 降级到默认值

        try:
            # 编码内容和高质量话题
            content_embedding = self._semantic_model.encode(content)
            topic_embeddings = self._semantic_model.encode(self._high_quality_topics)

            # 计算与所有话题的相似度
            similarities = self._semantic_util.cos_sim(content_embedding, topic_embeddings)
            max_similarity = float(similarities.max())

            # 检查否定词
            negative_score = self._check_negative_content(content)

            # 综合评分
            final_score = max(0.2, min(1.0, max_similarity - negative_score))
            return final_score

        except Exception as e:
            logger.error(f"语义评分失败: {e}")
            return 0.5

    def _check_negative_content(self, content: str) -> float:
        """检测否定内容，返回扣分值"""
        negative_count = 0
        for pattern in self._negative_patterns:
            if pattern.search(content):
                negative_count += 1

        return min(negative_count * 0.15, 0.3)  # 最多扣0.3分

    def _score_text_length(self, text: str) -> float:
        """改进的字幕长度评分（更宽松）"""
        actual_length = len(text.replace(' ', '').replace('\n', ''))

        optimal_min = self.config["length_optimal_min"]
        optimal_max = self.config["length_optimal_max"]
        good_min = self.config["length_good_min"]
        good_max = self.config["length_good_max"]

        if optimal_min <= actual_length <= optimal_max:
            return 1.0
        elif good_min <= actual_length < optimal_min:
            ratio = (actual_length - good_min) / (optimal_min - good_min)
            return 0.6 + 0.4 * ratio
        elif optimal_max < actual_length <= good_max:
            ratio = (good_max - actual_length) / (good_max - optimal_max)
            return 0.6 + 0.4 * ratio
        else:
            if actual_length < good_min:
                return max(0.3, (actual_length / good_min) * 0.6)
            else:
                return max(0.3, (good_max / actual_length) * 0.6)

    def _score_audio_energy(self, segment_index: int, energy_cache: Dict) -> float:
        """改进的音频能量评分（权重降低）"""
        energy = energy_cache.get(segment_index, 0.5)

        if 0.2 <= energy <= 0.8:
            return 0.7 + (min(energy, 0.5) - 0.2) / 0.3 * 0.3
        elif energy < 0.2:
            return 0.4 + energy / 0.2 * 0.3
        else:
            return 0.4 + (1.0 - energy) / 0.2 * 0.3

    def _score_vocabulary_diversity(self, text: str) -> float:
        """词汇多样性评分"""
        if not text:
            return 0.3

        clean_text = re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9]', '', text)
        if len(clean_text) == 0:
            return 0.3

        unique_chars = len(set(clean_text))
        total_chars = len(clean_text)
        diversity_ratio = unique_chars / total_chars if total_chars > 0 else 0

        if diversity_ratio >= 0.5:
            return 0.8 + (diversity_ratio - 0.5) / 0.5 * 0.2
        elif diversity_ratio >= 0.3:
            return 0.5 + (diversity_ratio - 0.3) / 0.2 * 0.3
        else:
            return max(0.2, diversity_ratio / 0.3 * 0.5)

    def _score_keywords(self, text: str) -> float:
        """改进的关键词检测评分"""
        keyword_patterns = [
            # 数字+单位（更精确）
            r'\d+(\.\d+)?%', r'\d+倍', r'\d+(年|月|日)',
            r'\d+万|\d+亿', r'\d+(元|美元|欧元)',

            # 强调词
            r'核心|关键|重要|必须|应当|建议|推荐',

            # 分析词
            r'分析|研究|发现|结论|观点|表明|显示',

            # 方法词
            r'方法|技巧|策略|步骤|流程|算法|模型',

            # 专业术语（新增）
            r'原理|机制|架构|框架|系统|平台'
        ]

        keyword_count = 0
        for pattern in keyword_patterns:
            if re.search(pattern, text):
                keyword_count += 1

        boost = min(keyword_count * 0.06, 0.3)
        return 0.4 + boost

    def score_clips(self, srt_data: List[Dict[str, Any]], energy_cache: Dict = None) -> List[EnhancedScoredClip]:
        """增强版评分方法"""
        if not srt_data:
            logger.warning("字幕数据为空")
            return []

        energy_cache = energy_cache or {}
        scored_clips = []

        for i, segment in enumerate(srt_data):
            content = segment.get('content', '') or segment.get('text', '')
            start_time = segment.get('start_time', segment.get('start', '00:00:00'))
            end_time = segment.get('end_time', segment.get('end', '00:00:00'))

            # 计算各维度得分
            semantic_score = self._score_semantic_relevance(content)
            length_score = self._score_text_length(content)
            energy_score = self._score_audio_energy(i, energy_cache)
            diversity_score = self._score_vocabulary_diversity(content)
            keyword_score = self._score_keywords(content)

            # 综合评分（加权平均）
            final_score = (
                semantic_score * self.config["semantic_weight"] +
                length_score * self.config["length_weight"] +
                energy_score * self.config["energy_weight"] +
                diversity_score * self.config["diversity_weight"] +
                keyword_score * self.config["keyword_weight"]
            )

            final_score = max(0.25, min(1.0, final_score))

            # 计算置信度（基于特征一致性）
            feature_scores = [semantic_score, length_score, energy_score, diversity_score, keyword_score]
            confidence = 1.0 - (max(feature_scores) - min(feature_scores)) / 2

            scored = EnhancedScoredClip(
                id=i,
                start_time=start_time,
                end_time=end_time,
                content=content if content else f"片段{i+1}内容",
                final_score=round(final_score, 3),
                scoring_method="enhanced_local",
                quality_note="[INFO] 增强版本地评分（含语义理解）",
                features={
                    "semantic_score": round(semantic_score, 3),
                    "length_score": round(length_score, 3),
                    "energy_score": round(energy_score, 3),
                    "diversity_score": round(diversity_score, 3),
                    "keyword_score": round(keyword_score, 3)
                },
                confidence=round(confidence, 3)
            )

            scored_clips.append(scored)

        logger.info(f"增强版本地评分完成，分数范围: {min(c.final_score for c in scored_clips):.3f} - {max(c.final_score for c in scored_clips):.3f}")
        return scored_clips
```

### 5.2 自适应阈值机制（原2.2）

#### 核心思想

根据分数分布动态调整筛选阈值，避免固定阈值的不适应性。

```python
"""
自适应阈值筛选器
"""
import statistics
from typing import List, Dict, Any
import logging

logger = logging.getLogger(__name__)


class AdaptiveThresholdFilter:
    """自适应阈值筛选器"""

    def __init__(self):
        self.config = {
            "min_clips_to_keep": 5,    # 最少保留数量
            "max_clips_to_keep": 20,   # 最多保留数量
            "base_threshold": 0.5,     # 基础阈值
            "confidence_weight": 0.3,  # 置信度权重
        }

    def calculate_adaptive_threshold(self, clips: List[Dict[str, Any]]) -> float:
        """根据分数分布计算自适应阈值"""
        if not clips:
            return self.config["base_threshold"]

        scores = [clip.get('final_score', 0) for clip in clips]
        confidences = [clip.get('confidence', 0.5) for clip in clips]

        # 计算分数统计特征
        mean_score = sum(scores) / len(scores)
        std_score = statistics.stdev(scores) if len(scores) > 1 else 0.1

        # 计算置信度加权分数
        weighted_scores = [
            scores[i] * (0.7 + confidences[i] * 0.3)
            for i in range(len(scores))
        ]
        weighted_mean = sum(weighted_scores) / len(weighted_scores)

        # 自适应阈值 = 加权平均分 - 标准差的一半
        adaptive_threshold = weighted_mean - std_score * 0.5

        # 限制阈值范围
        return max(
            self.config["base_threshold"],
            min(0.85, adaptive_threshold)
        )

    def filter_clips(self, clips: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """使用自适应阈值筛选切片"""
        if not clips:
            return []

        # 计算自适应阈值
        threshold = self.calculate_adaptive_threshold(clips)
        logger.info(f"自适应阈值计算结果: {threshold:.3f}")

        # 按分数筛选
        high_score_clips = [
            clip for clip in clips
            if clip.get('final_score', 0) >= threshold
        ]

        # 保底机制（改进版）
        if len(high_score_clips) < self.config["min_clips_to_keep"]:
            remaining_clips = [c for c in clips if c not in high_score_clips]
            remaining_clips.sort(key=lambda x: x.get('confidence', 0.5) * x.get('final_score', 0), reverse=True)

            needed = self.config["min_clips_to_keep"] - len(high_score_clips)
            high_score_clips.extend(remaining_clips[:needed])

            logger.info(f"保底补充: {needed} 个切片")

        # 上限控制
        if len(high_score_clips) > self.config["max_clips_to_keep"]:
            high_score_clips.sort(key=lambda x: x.get('final_score', 0), reverse=True)
            high_score_clips = high_score_clips[:self.config["max_clips_to_keep"]]

        return high_score_clips
```

### 5.3 混合评分融合策略（原2.3）

#### 核心思想

结合LLM和本地评分的优势，实现优雅的混合评分。

```python
"""
混合评分融合器
"""
from typing import List, Dict, Any, Optional
import logging

logger = logging.getLogger(__name__)


class HybridScoringFusion:
    """混合评分融合器"""

    def __init__(self):
        self.config = {
            "llm_weight": 0.6,      # LLM评分权重
            "local_weight": 0.4,     # 本地评分权重
            "llm_confidence_threshold": 0.7,  # LLM置信度阈值
        }

    def _estimate_llm_confidence(self, llm_result: Dict[str, Any]) -> float:
        """估算LLM评分的置信度"""
        has_score = 'final_score' in llm_result and llm_result['final_score'] is not None
        has_reason = 'recommend_reason' in llm_result and llm_result['recommend_reason']
        score_in_range = 0 <= llm_result.get('final_score', 0) <= 1

        confidence = 0.5
        if has_score:
            confidence += 0.2
        if has_reason and len(llm_result['recommend_reason']) > 10:
            confidence += 0.2
        if score_in_range:
            confidence += 0.1

        return min(1.0, confidence)

    def fuse_scores(
        self,
        llm_scores: Optional[List[Dict[str, Any]]],
        local_scores: List[Dict[str, Any]]
    ) -> List[Dict[str, Any]]:
        """融合LLM和本地评分"""
        result = []

        if not llm_scores:
            # LLM完全不可用，使用本地评分但标记
            for local in local_scores:
                local['score_source'] = 'local_only'
                local['fusion_note'] = "LLM不可用，使用本地评分"
                result.append(local)
            return result

        # 对齐数据
        for i, local in enumerate(local_scores):
            llm = llm_scores[i] if i < len(llm_scores) else None

            if llm:
                llm_confidence = self._estimate_llm_confidence(llm)

                # 根据LLM置信度动态调整权重
                if llm_confidence >= self.config["llm_confidence_threshold"]:
                    weight_llm = min(0.8, self.config["llm_weight"] + (llm_confidence - 0.7) * 0.5)
                    weight_local = 1 - weight_llm
                else:
                    weight_llm = max(0.4, self.config["llm_weight"] - (0.7 - llm_confidence) * 0.5)
                    weight_local = 1 - weight_llm

                # 融合评分
                llm_score = llm.get('final_score', 0)
                local_score = local.get('final_score', 0)

                final_score = llm_score * weight_llm + local_score * weight_local

                fused = {
                    **local,
                    'final_score': round(final_score, 3),
                    'llm_score': llm_score,
                    'local_score': local_score,
                    'score_source': 'hybrid',
                    'fusion_weights': {'llm': weight_llm, 'local': weight_local},
                    'llm_confidence': llm_confidence,
                    'fusion_note': f"LLM置信度: {llm_confidence:.2f}"
                }

                result.append(fused)
            else:
                local['score_source'] = 'local_fallback'
                local['fusion_note'] = "LLM结果不足，使用本地评分"
                result.append(local)

        logger.info(f"混合评分融合完成，共 {len(result)} 个片段")
        return result
```
