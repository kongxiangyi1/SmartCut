## 一、项目概述

### 1.1 背景

当前评分流程存在以下关键问题：

1. **系统可靠性不足**：LLM评分失败时整个流程中断
2. **阈值策略僵化**：固定阈值0.7无法适应不同内容
3. **重要内容丢失**：核心话题可能被错误过滤

### 1.2 目标

通过实施优化方案组合（方案A + 方案C + 方案E），实现：

- ✅ 系统可用性提升至99.9%
- ✅ 重要内容召回率提升20%
- ✅ 阈值自适应调整，适应不同内容
- ✅ 多层降级机制，保证流程稳定

### 1.3 范围

**优化范围**：
- 修改文件：`backend/pipeline/step3_scoring.py`
- 新增文件：`backend/pipeline/scoring_fallback.py`（可选）

**影响模块**：
- 评分流程（Step 3）
- 依赖该流程的所有上游模块

### 4.1 方案A：系统可靠性提升

#### 4.1.1 目标

#### 4.1.2 实现方案

**文件**：`backend/pipeline/step3_scoring.py`

**新增方法**：

```python
class ClipScorer:
    def score_clips_with_fallback(self, timeline_data: List[Dict]) -> List[Dict]:
        """
        评分方法 - 多层降级版本
        
        降级策略：
        1. 优先使用LLM评分
        2. LLM失败降级到本地评分
        3. 本地失败降级到默认评分
        
        Returns:
            List[Dict]: 评分后的切片列表
            
        Raises:
            Exception: 仅在所有评分方式都失败时抛出
        """
        if not timeline_data:
            logger.warning("时间线数据为空，跳过评分")
            return []
        
        # 记录开始时间
        start_time = time.time()
        
        # 评分来源追踪
        scoring_history = {
            "total_clips": len(timeline_data),
            "attempts": []
        }
        
        # 第一层：LLM评分
        try:
            logger.info("尝试使用LLM评分...")
            scored_clips = self._score_with_llm(timeline_data)
            
            scoring_history["attempts"].append({
                "method": "llm",
                "status": "success",
                "count": len(scored_clips),
                "time": time.time() - start_time
            })
            
            logger.info(f"LLM评分成功: {len(scored_clips)} 个切片")
            return scored_clips
            
        except Exception as llm_error:
            logger.error(f"LLM评分失败: {str(llm_error)}")
            
            scoring_history["attempts"].append({
                "method": "llm",
                "status": "failed",
                "error": str(llm_error)
            })
        
        # 第二层：本地评分
        try:
            logger.info("降级到本地评分...")
            scored_clips = self._score_with_local_fallback(timeline_data)
            
            scoring_history["attempts"].append({
                "method": "local",
                "status": "success",
                "count": len(scored_clips),
                "time": time.time() - start_time
            })
            
            logger.info(f"本地评分成功: {len(scored_clips)} 个切片")
            return scored_clips
            
        except Exception as local_error:
            logger.error(f"本地评分失败: {str(local_error)}")
            
            scoring_history["attempts"].append({
                "method": "local",
                "status": "failed",
                "error": str(local_error)
            })
        
        # 第三层：默认评分（兜底）
        logger.warning("所有评分方式失败，使用默认评分兜底")
        scored_clips = self._score_with_default_fallback(timeline_data)
        
        scoring_history["attempts"].append({
            "method": "default",
            "status": "success",
            "count": len(scored_clips),
            "time": time.time() - start_time
        })
        
        logger.info(f"默认评分完成: {len(scored_clips)} 个切片")
        
        # 记录评分历史
        self._log_scoring_history(scoring_history)
        
        return scored_clips
    
    def _score_with_local_fallback(self, timeline_data: List[Dict]) -> List[Dict]:
        """
        本地评分降级方法
        
        使用LocalScorer进行评分，如果不可用则抛出异常
        
        Returns:
            List[Dict]: 评分后的切片列表
            
        Raises:
            Exception: 本地评分失败时抛出
        """
        try:
            from backend.utils.local_scorer import LocalScorer
            
            # 构建SRT格式数据
            srt_data = self._prepare_srt_data(timeline_data)
            
            # 使用本地评分器
            scorer = LocalScorer()
            scored_results = scorer.score_clips(srt_data)
            
            # 映射回原格式
            for i, clip in enumerate(timeline_data):
                if i < len(scored_results):
                    clip['final_score'] = scored_results[i].final_score
                    clip['recommend_reason'] = "本地评分（LLM降级）"
                    clip['score_source'] = 'local'
                else:
                    clip['final_score'] = 0.7
                    clip['recommend_reason'] = "自动评估"
                    clip['score_source'] = 'local'
            
            return timeline_data
            
        except ImportError:
            raise Exception("LocalScorer不可用")
        except Exception as e:
            raise Exception(f"本地评分异常: {str(e)}")
    
    def _score_with_default_fallback(self, timeline_data: List[Dict]) -> List[Dict]:
        """
        默认评分兜底方法
        
        使用基于内容的启发式评分，确保每个切片都有评分
        
        Returns:
            List[Dict]: 评分后的切片列表
        """
        import random
        
        for clip in timeline_data:
            # 确保content存在
            if not clip.get('content'):
                outline = clip.get('outline', '')
                if isinstance(outline, dict):
                    content = outline.get('title', '') or outline.get('content', '')
                else:
                    content = str(outline)
                clip['content'] = content if content else "片段内容"
            
            # 计算内容质量分数
            content_length = len(clip.get('content', ''))
            
            # 基础分数：0.6-0.8
            base_score = 0.6 + random.random() * 0.2
            
            # 长度调整：适中长度加分
            if 50 <= content_length <= 200:
                base_score += 0.1
            elif content_length < 20:
                base_score -= 0.2
            
            clip['final_score'] = round(max(0.3, min(0.9, base_score)), 2)
            clip['recommend_reason'] = "默认评分（系统降级）"
            clip['score_source'] = 'default'
        
        return timeline_data
    
    def _prepare_srt_data(self, timeline_data: List[Dict]) -> List[Dict]:
        """准备SRT格式数据"""
        srt_data = []
        
        for i, clip in enumerate(timeline_data):
            content = clip.get('content', '')
            if not content:
                outline = clip.get('outline', '')
                if isinstance(outline, dict):
                    content = outline.get('title', '') or outline.get('content', '')
                else:
                    content = str(outline)
            
            srt_data.append({
                'index': i + 1,
                'start_time': clip.get('start_time', '00:00:00'),
                'end_time': clip.get('end_time', '00:00:10'),
                'content': content if content else f"片段{i+1}内容"
            })
        
        return srt_data
    
    def _log_scoring_history(self, history: Dict):
        """记录评分历史"""
        logger.info("=== 评分历史记录 ===")
        logger.info(f"总切片数: {history['total_clips']}")
        
        for attempt in history['attempts']:
            method = attempt['method'].upper()
            status = attempt['status']
            
            if status == 'success':
                logger.info(f"  [{method}] 成功: {attempt['count']}个, 耗时: {attempt['time']:.2f}秒")
            else:
                logger.info(f"  [{method}] 失败: {attempt.get('error', '未知错误')}")
```

### 4.2 方案C：自适应阈值机制

#### 4.2.1 目标
根据分数分布动态计算筛选阈值，避免固定阈值的不适应性

#### 4.2.2 实现方案

**新增类**：`AdaptiveThresholdFilter`

```python
import math

class AdaptiveThresholdFilter:
    """
    自适应阈值筛选器
    
    根据分数分布自动计算最优阈值，确保：
    1. 保留足够数量的切片（保底机制）
    2. 限制最大数量（避免过多）
    3. 阈值在合理范围内
    """
    
    def __init__(self, config: Dict = None):
        """
        初始化自适应阈值筛选器
        
        Args:
            config: 可选配置字典
        """
        self.config = config or {
            "min_clips": 5,           # 最少保留数量
            "max_clips": 20,          # 最多保留数量
            "base_threshold": 0.5,    # 基础阈值
            "min_threshold": 0.4,     # 最小阈值
            "max_threshold": 0.85,    # 最大阈值
            "use_percentile": True,   # 使用百分位数
            "percentile": 0.7          # 保留前70%
        }
    
    def calculate_threshold(self, clips: List[Dict]) -> float:
        """
        根据分数分布计算自适应阈值
        
        算法：
        1. 收集所有分数
        2. 计算统计特征（均值、标准差）
        3. 使用自适应公式计算阈值
        4. 限制在合理范围内
        
        Args:
            clips: 评分后的切片列表
            
        Returns:
            float: 计算得到的阈值
        """
        if not clips:
            return self.config["base_threshold"]
        
        # 收集分数
        scores = [clip.get('final_score', 0) for clip in clips]
        
        # 计算统计特征
        n = len(scores)
        mean_score = sum(scores) / n
        
        # 计算标准差
        variance = sum((s - mean_score) ** 2 for s in scores) / n
        std_score = math.sqrt(variance) if variance > 0 else 0.1
        
        # 自适应阈值计算
        # 公式: mean - 0.5 * std
        # 这样高分集中的数据阈值会较高，低分分散的数据阈值会较低
        adaptive_threshold = mean_score - std_score * 0.5
        
        # 百分位数调整
        if self.config["use_percentile"]:
            sorted_scores = sorted(scores)
            percentile_index = int(n * self.config["percentile"])
            percentile_score = sorted_scores[percentile_index]
            
            # 综合自适应阈值和百分位数
            adaptive_threshold = (adaptive_threshold * 0.6 + percentile_score * 0.4)
        
        # 应用置信度调整（如果有）
        if any('confidence' in clip for clip in clips):
            confidences = [clip.get('confidence', 0.5) for clip in clips]
            confidence_factor = sum(confidences) / len(confidences)
            
            # 高置信度数据可以提高阈值
            adaptive_threshold = adaptive_threshold * (0.9 + confidence_factor * 0.1)
        
        # 限制范围
        final_threshold = max(
            self.config["min_threshold"],
            min(self.config["max_threshold"], adaptive_threshold)
        )
        
        # 记录日志
        logger.info(
            f"自适应阈值计算: 均值={mean_score:.3f}, 标准差={std_score:.3f}, "
            f"计算阈值={adaptive_threshold:.3f}, 最终阈值={final_threshold:.3f}"
        )
        
        return round(final_threshold, 3)
    
    def filter_clips(self, clips: List[Dict]) -> List[Dict]:
        """
        使用自适应阈值筛选切片
        
        Args:
            clips: 评分后的切片列表
            
        Returns:
            List[Dict]: 筛选后的切片列表
        """
        if not clips:
            return []
        
        # 计算自适应阈值
        threshold = self.calculate_threshold(clips)
        logger.info(f"使用自适应阈值 {threshold:.3f} 筛选切片")
        
        # 按分数排序
        sorted_clips = sorted(clips, key=lambda x: x.get('final_score', 0), reverse=True)
        
        # 第一步：按阈值筛选
        high_score_clips = [
            clip for clip in sorted_clips
            if clip.get('final_score', 0) >= threshold
        ]
        
        logger.info(f"阈值筛选后: {len(high_score_clips)} 个切片")
        
        # 第二步：保底机制 - 确保最少保留数量
        if len(high_score_clips) < self.config["min_clips"]:
            logger.warning(
                f"筛选结果过少({len(high_score_clips)} < {self.config['min_clips']})，"
                f"自动补充低分切片"
            )
            
            # 补充剩余切片中分数最高的
            remaining_clips = [c for c in sorted_clips if c not in high_score_clips]
            needed = self.config["min_clips"] - len(high_score_clips)
            
            high_score_clips.extend(remaining_clips[:needed])
            logger.info(f"保底补充后: {len(high_score_clips)} 个切片")
        
        # 第三步：上限控制 - 限制最大数量
        if len(high_score_clips) > self.config["max_clips"]:
            logger.info(
                f"筛选结果过多({len(high_score_clips)} > {self.config['max_clips']})，"
                f"截断至前{self.config['max_clips']}个"
            )
            high_score_clips = high_score_clips[:self.config["max_clips"]]
        
        # 按原始顺序排序
        high_score_clips.sort(key=lambda x: int(x.get('id', 0)))
        
        # 记录统计信息
        scores = [c.get('final_score', 0) for c in high_score_clips]
        logger.info(
            f"筛选完成: {len(high_score_clips)} 个切片, "
            f"分数范围: {min(scores):.3f} - {max(scores):.3f}"
        )
        
        return high_score_clips
    
    def get_filtering_stats(self, clips: List[Dict], filtered_clips: List[Dict]) -> Dict:
        """
        获取筛选统计信息
        
        Returns:
            Dict: 统计信息字典
        """
        if not clips or not filtered_clips:
            return {}
        
        all_scores = [c.get('final_score', 0) for c in clips]
        filtered_scores = [c.get('final_score', 0) for c in filtered_clips]
        
        return {
            "total_clips": len(clips),
            "filtered_clips": len(filtered_clips),
            "filter_ratio": len(filtered_clips) / len(clips),
            "all_score_range": f"{min(all_scores):.3f} - {max(all_scores):.3f}",
            "filtered_score_range": f"{min(filtered_scores):.3f} - {max(filtered_scores):.3f}",
            "avg_score_increase": sum(filtered_scores) / len(filtered_scores) - sum(all_scores) / len(all_scores)
        }
```

### 4.3 方案E：重要性识别

#### 4.3.1 目标
识别重要话题，确保重要内容不被过滤

#### 4.3.2 实现方案

**新增类**：`ImportanceIdentifier`

```python
class ImportanceIdentifier:
    """
    重要性识别器
    
    通过关键词匹配识别重要话题，并调整筛选阈值
    """
    
    def __init__(self, custom_keywords: Dict = None):
        """
        初始化重要性识别器
        
        Args:
            custom_keywords: 自定义关键词字典
        """
        # 默认重要关键词分类
        self.important_keywords = custom_keywords or {
            "core": {
                "keywords": ["核心", "关键", "重要", "必须", "唯一", "首要", "重点"],
                "threshold_adjustment": -0.15,
                "description": "核心/关键内容"
            },
            "product": {
                "keywords": ["产品", "销售", "介绍", "优惠", "促销", "讲解", "推荐"],
                "threshold_adjustment": -0.10,
                "description": "产品/销售相关"
            },
            "data": {
                "keywords": ["分析", "研究", "数据", "结论", "发现", "表明", "显示", "统计"],
                "threshold_adjustment": -0.10,
                "description": "数据分析相关"
            },
            "method": {
                "keywords": ["方法", "技巧", "策略", "步骤", "流程", "算法", "模型", "公式"],
                "threshold_adjustment": -0.10,
                "description": "方法/技巧相关"
            },
            "warning": {
                "keywords": ["注意", "警告", "警惕", "风险", "隐患", "禁止", "切勿"],
                "threshold_adjustment": -0.15,
                "description": "警告/提醒相关"
            },
            "action": {
                "keywords": ["操作", "执行", "运行", "启动", "设置", "配置", "安装"],
                "threshold_adjustment": -0.08,
                "description": "操作指导相关"
            }
        }
        
        # 编译正则表达式以提高效率
        self._compiled_patterns = {}
        for category, config in self.important_keywords.items():
            keywords = config["keywords"]
            pattern = "|".join([re.escape(kw) for kw in keywords])
            self._compiled_patterns[category] = re.compile(pattern)
    
    def identify_importance(self, clip: Dict) -> Dict:
        """
        识别切片的重要性
        
        Args:
            clip: 切片数据字典
            
        Returns:
            Dict: 重要性识别结果
                {
                    "is_important": bool,
                    "importance_type": str,
                    "keywords_found": List[str],
                    "threshold_adjustment": float,
                    "description": str
                }
        """
        # 合并content和outline进行关键词匹配
        content = clip.get('content', '') or ''
        outline = clip.get('outline', '')
        
        if isinstance(outline, dict):
            outline_text = outline.get('title', '') + ' ' + outline.get('content', '')
        else:
            outline_text = str(outline)
        
        full_text = (content + ' ' + outline_text).lower()
        
        # 关键词匹配
        for category, config in self.important_keywords.items():
            pattern = self._compiled_patterns[category]
            matches = pattern.findall(full_text)
            
            if matches:
                return {
                    "is_important": True,
                    "importance_type": category,
                    "keywords_found": list(set(matches)),
                    "threshold_adjustment": config["threshold_adjustment"],
                    "description": config["description"]
                }
        
        # 没有匹配到重要关键词
        return {
            "is_important": False,
            "importance_type": None,
            "keywords_found": [],
            "threshold_adjustment": 0.0,
            "description": "普通内容"
        }
    
    def adjust_threshold_for_important(self, base_threshold: float, clip: Dict) -> float:
        """
        为重要话题调整阈值
        
        Args:
            base_threshold: 基础阈值
            clip: 切片数据
            
        Returns:
            float: 调整后的阈值
        """
        importance = self.identify_importance(clip)
        
        if importance["is_important"]:
            adjusted_threshold = base_threshold + importance["threshold_adjustment"]
            
            logger.info(
                f"重要话题识别: '{clip.get('outline', '')[:30]}...' "
                f"类型={importance['description']}, "
                f"关键词={importance['keywords_found']}, "
                f"阈值调整: {base_threshold:.3f} -> {adjusted_threshold:.3f}"
            )
            
            return max(0.3, adjusted_threshold)  # 最低不低于0.3
        
        return base_threshold
    
    def filter_with_importance(self, clips: List[Dict], base_threshold: float) -> List[Dict]:
        """
        结合重要性的智能筛选
        
        Args:
            clips: 切片列表
            base_threshold: 基础阈值
            
        Returns:
            List[Dict]: 筛选后的切片列表
        """
        if not clips:
            return []
        
        # 为每个切片计算调整后的阈值
        adjusted_clips = []
        important_count = 0
        
        for clip in clips:
            importance = self.identify_importance(clip)
            
            # 调整阈值
            adjusted_threshold = self.adjust_threshold_for_important(base_threshold, clip)
            clip['_adjusted_threshold'] = adjusted_threshold
            clip['_importance_info'] = importance
            
            adjusted_clips.append(clip)
            
            if importance["is_important"]:
                important_count += 1
        
        # 按调整后的阈值筛选
        filtered = [
            clip for clip in adjusted_clips
            if clip.get('final_score', 0) >= clip['_adjusted_threshold']
        ]
        
        logger.info(
            f"重要性筛选完成: {len(filtered)}/{len(clips)} 个切片, "
            f"重要切片: {important_count}"
        )
        
        return filtered
    
    def get_importance_stats(self, clips: List[Dict]) -> Dict:
        """
        获取重要性统计信息
        
        Returns:
            Dict: 统计信息
        """
        stats = {
            "total_clips": len(clips),
            "important_clips": 0,
            "by_category": {}
        }
        
        for clip in clips:
            importance = self.identify_importance(clip)
            if importance["is_important"]:
                stats["important_clips"] += 1
                category = importance["importance_type"]
                stats["by_category"][category] = stats["by_category"].get(category, 0) + 1
        
        stats["important_ratio"] = stats["important_clips"] / len(clips) if clips else 0
        
        return stats
```

def run_step3_scoring(timeline_path: Path, output_path: Path, config: Dict = None):
    """运行Step3评分的入口函数"""
    
    # 1. 加载时间线数据
    timeline_data = load_timeline(timeline_path)

    base_threshold = filter.calculate_threshold(scored_clips)
    
    # 结合重要性的智能筛选
    filtered_clips = importance_identifier.filter_with_importance(
        scored_clips, 
        base_threshold
    )
    
    # 5. 如果筛选结果过少，放宽阈值
    if len(filtered_clips) < filter.config["min_clips"]:
        logger.warning(f"筛选结果过少({len(filtered_clips)})，放宽阈值重试")
        filtered_clips = importance_identifier.filter_with_importance(
            scored_clips,
            base_threshold * 0.9  # 降低10%
        )
    
    # 6. 保存结果
    save_scores(filtered_clips, output_path)
    
    return filtered_clips
