## 三、方案A：系统可靠性提升

### 目标
即使LLM评分失败时，提供优雅的降级方案，保证流程能继续进行

### 核心思路
- **多层降级**：LLM → 本地评分 → 默认评分
- **异常捕获**：捕获所有可能的异常类型
- **错误日志**：详细记录失败原因

### 实现方案

#### 3.1 修改 `score_clips` 方法

```python
# 文件：backend/pipeline/step3_scoring.py

def score_clips(self, timeline_data: List[Dict]) -> List[Dict]:
    """
    为切片评分 - 增强版：多层降级机制
    """
    if not timeline_data:
        logger.warning("时间线数据为空，无法评分")
        raise Exception("时间线数据为空")
        
    logger.info(f"开始为 {len(timeline_data)} 个切片进行批量评分...")
    
    try:
        # 尝试LLM评分（主路径）
        return self._score_with_llm(timeline_data)
    except Exception as llm_error:
        logger.error(f"LLM评分失败: {str(llm_error)}")
        
        # 降级到本地评分
        try:
            logger.info("降级到本地评分器")
            return self._score_with_local(timeline_data)
        except Exception as local_error:
            logger.error(f"本地评分也失败: {str(local_error)}")
            
            # 最终降级到默认评分
            logger.info("使用默认评分兜底")
            return self._score_with_default(timeline_data)
```

#### 3.2 新增 `_score_with_llm` 方法

```python
def _score_with_llm(self, timeline_data: List[Dict]) -> List[Dict]:
    """使用LLM进行评分"""
    timeline_by_chunk = defaultdict(list)
    no_chunk_items = []
    
    for item in timeline_data:
        chunk_index = item.get('chunk_index')
        if chunk_index is not None:
            timeline_by_chunk[chunk_index].append(item)
        else:
            no_chunk_items.append(item)
    
    all_scored_clips = []
    
    # 处理有chunk_index的项
    for chunk_index, chunk_items in timeline_by_chunk.items():
        scored_chunk_items = self._get_llm_evaluation(chunk_items)
        if scored_chunk_items:
            all_scored_clips.extend(scored_chunk_items)
        else:
            raise Exception(f"块 {chunk_index} 的LLM评估返回为空")
    
    # 处理无chunk_index的项
    if no_chunk_items:
        scored_items = self._get_llm_evaluation(no_chunk_items)
        if scored_items:
            all_scored_clips.extend(scored_items)
        else:
            raise Exception("无chunk_index项的LLM评估返回为空")
    
    # 保持原有顺序
    all_scored_clips.sort(key=lambda x: int(x.get('id', 0)))
    return all_scored_clips
```

```python
def _score_with_local(self, timeline_data: List[Dict]) -> List[Dict]:
    """使用本地评分器降级评分"""
    try:
        from backend.utils.local_scorer import LocalScorer
        
        # 构建SRT格式数据
        srt_data = []
        for i, clip in enumerate(timeline_data):
            content = clip.get('content', '')
            if not content:
                outline = clip.get('outline', '')
                if isinstance(outline, dict):
                    content = outline.get('title', '') or outline.get('content', '')
                else:
                    content = str(outline)
            if not content:
                content = f"片段{i+1}内容"
            
            srt_data.append({
                'index': i + 1,
                'start_time': clip.get('start_time', '00:00:00'),
                'end_time': clip.get('end_time', '00:00:10'),
                'content': content
            })
        
        scorer = LocalScorer()
        scored_clips = scorer.score_clips(srt_data)
        
        # 映射回原格式
        for i, clip in enumerate(timeline_data):
            if i < len(scored_clips):
                clip['final_score'] = scored_clips[i].final_score
                clip['recommend_reason'] = "本地评分（LLM降级）"
                clip['score_source'] = 'local_fallback'
            else:
                clip['final_score'] = 0.7
                clip['recommend_reason'] = "自动评估"
                clip['score_source'] = 'local_fallback'
        
        logger.info(f"本地评分完成，共 {len(timeline_data)} 个片段")
        return timeline_data
        
    except Exception as e:
        logger.error(f"本地评分器使用失败: {e}")
        raise
```

#### 3.4 新增 `_score_with_default` 方法

```python
def _score_with_default(self, timeline_data: List[Dict]) -> List[Dict]:
    """使用默认评分兜底"""
    import random
    
    for clip in timeline_data:
        # 确保content存在
        if 'content' not in clip or not clip.get('content'):
            outline = clip.get('outline', '')
            if isinstance(outline, dict):
                content = outline.get('title', '') or outline.get('content', '')
            else:
                content = str(outline)
            clip['content'] = content if content else "片段内容"
        
        # 给一个合理的随机分数（0.5-0.9之间）
        clip['final_score'] = round(0.5 + random.random() * 0.4, 2)
        clip['recommend_reason'] = "自动评估（LLM和本地评分均失败）"
        clip['score_source'] = 'default_fallback'
    
    logger.info(f"默认评分完成，共 {len(timeline_data)} 个片段")
    return timeline_data
```

## 五、方案C：自适应阈值机制

### 目标
根据分数分布动态调整筛选阈值，避免固定阈值的不适应性

### 核心思路
- **数据分析**：分析分数分布特征（均值、标准差）
- **动态计算**：根据分布自动计算合适的阈值
- **上下限保护**：设置合理的阈值范围

### 实现方案

#### 5.1 新增自适应阈值筛选器

```python
# 文件：backend/pipeline/step3_scoring.py

class AdaptiveThresholdFilter:
    """自适应阈值筛选器"""
    
    def __init__(self):
        self.config = {
            "min_clips_to_keep": 5,      # 最少保留数量
            "max_clips_to_keep": 20,     # 最多保留数量
            "base_threshold": 0.5,       # 基础阈值
            "min_threshold": 0.4,        # 最小阈值
            "max_threshold": 0.85,       # 最大阈值
            "confidence_weight": 0.3     # 置信度权重
        }
    
    def calculate_adaptive_threshold(self, clips: List[Dict[str, Any]]) -> float:
        """根据分数分布计算自适应阈值"""
        if not clips:
            return self.config["base_threshold"]
        
        scores = [clip.get('final_score', 0) for clip in clips]
        
        # 计算统计特征
        mean_score = sum(scores) / len(scores)
        variance = sum((s - mean_score) ** 2 for s in scores) / len(scores)
        std_score = math.sqrt(variance) if variance > 0 else 0.1
        
        # 自适应阈值 = 平均分 - 标准差的一半
        # 这样可以根据数据分布自动调整
        adaptive_threshold = mean_score - std_score * 0.5
        
        # 应用置信度调整（如果有置信度信息）
        if any('confidence' in clip for clip in clips):
            confidences = [clip.get('confidence', 0.5) for clip in clips]
            weighted_mean = sum(
                scores[i] * (0.7 + confidences[i] * 0.3)
                for i in range(len(scores))
            ) / len(scores)
            adaptive_threshold = adaptive_threshold * 0.7 + weighted_mean * 0.3
        
        # 限制阈值范围
        return max(
            self.config["min_threshold"],
            min(self.config["max_threshold"], adaptive_threshold)
        )
    
    def filter_clips(self, clips: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """使用自适应阈值筛选切片"""
        if not clips:
            return []
        
        # 计算自适应阈值
        threshold = self.calculate_adaptive_threshold(clips)
        logger.info(f"自适应阈值计算结果: {threshold:.3f}")
        
        # 按分数筛选
        high_score_clips = [
            clip for clip in clips
            if clip.get('final_score', 0) >= threshold
        ]
        
        # 保底机制
        if len(high_score_clips) < self.config["min_clips_to_keep"]:
            remaining_clips = [c for c in clips if c not in high_score_clips]
            remaining_clips.sort(key=lambda x: x.get('final_score', 0), reverse=True)
            
            needed = self.config["min_clips_to_keep"] - len(high_score_clips)
            high_score_clips.extend(remaining_clips[:needed])
            logger.info(f"保底补充: {needed} 个切片")
        
        # 上限控制
        if len(high_score_clips) > self.config["max_clips_to_keep"]:
            high_score_clips.sort(key=lambda x: x.get('final_score', 0), reverse=True)
            high_score_clips = high_score_clips[:self.config["max_clips_to_keep"]]
        
        return high_score_clips
```

- **结果验证**：确保输出符合预期

### 实现方案

#### 6.1 新增质量保证层

```python
# 文件：backend/pipeline/step3_scoring.py

class QualityAssurance:
    """质量保证层"""
    
    def __init__(self):
        self.config = {
            "min_diversity_score": 0.3,        # 最小多样性分数
            "max_similarity_allowed": 0.8,     # 最大相似度允许值
            "min_content_length": 10,          # 最小内容长度
            "max_content_length": 500          # 最大内容长度
        }
    
    def check_diversity(self, clips: List[Dict]) -> List[Dict]:
        """检查并确保结果多样性"""
        if len(clips) <= 1:
            return clips
        
        # 使用简单的文本相似度检查
        unique_clips = []
        for i, clip_i in enumerate(clips):
            is_unique = True
            content_i = clip_i.get('content', '')[:100]
            
            for clip_j in unique_clips:
                content_j = clip_j.get('content', '')[:100]
                similarity = self._calculate_text_similarity(content_i, content_j)
                
                if similarity > self.config["max_similarity_allowed"]:
                    is_unique = False
                    logger.info(f"跳过相似内容: {clip_i.get('outline', '')[:20]}...")
                    break
            
            if is_unique:
                unique_clips.append(clip_i)
        
        return unique_clips
    
    def _calculate_text_similarity(self, text1: str, text2: str) -> float:
        """计算文本相似度（Jaccard相似度）"""
        words1 = set(text1.lower().split())
        words2 = set(text2.lower().split())
        
        if not words1 and not words2:
            return 0.0
        
        intersection = len(words1 & words2)
        union = len(words1 | words2)
        
        return intersection / union if union > 0 else 0.0
    
    def validate_content(self, clip: Dict) -> bool:
        """验证单个内容的质量"""
        content = clip.get('content', '')
        
        # 检查内容长度
        if len(content) < self.config["min_content_length"]:
            return False
        if len(content) > self.config["max_content_length"]:
            return False
        
        # 检查是否有有效内容
        if not content.strip():
            return False
        
        return True
```

```python
# 文件：backend/pipeline/step3_scoring.py

class ImportanceIdentifier:
    """重要性识别器"""
    
    def __init__(self):
        # 重要关键词列表
        self.important_keywords = {
            "core": ["核心", "关键", "重要", "必须", "唯一", "首要"],
            "product": ["产品", "销售", "介绍", "优惠", "促销", "讲解", "推荐"],
            "data": ["分析", "研究", "数据", "结论", "发现", "表明", "显示"],
            "method": ["方法", "技巧", "策略", "步骤", "流程", "算法", "模型"],
            "warning": ["注意", "警告", "警惕", "风险", "隐患"]
        }
        
        # 重要话题类型及其阈值降低幅度
        self.topic_adjustments = {
            "core": -0.15,    # 核心话题降低15%阈值
            "product": -0.10, # 产品话题降低10%阈值
            "data": -0.10,    # 数据分析话题降低10%阈值
            "method": -0.10,  # 方法技巧话题降低10%阈值
            "warning": -0.15  # 警告提醒话题降低15%阈值
        }
    
    def identify_importance(self, clip: Dict) -> Dict:
        """识别话题的重要性"""
        content = clip.get('content', '') + ' ' + str(clip.get('outline', ''))
        
        importance_info = {
            "is_important": False,
            "importance_type": None,
            "keywords_found": [],
            "threshold_adjustment": 0.0
        }
        
        # 检查各类型关键词
        for topic_type, keywords in self.important_keywords.items():
            found_keywords = [kw for kw in keywords if kw in content]
            if found_keywords:
                importance_info["is_important"] = True
                importance_info["importance_type"] = topic_type
                importance_info["keywords_found"].extend(found_keywords)
                importance_info["threshold_adjustment"] = self.topic_adjustments.get(topic_type, 0.0)
                break  # 只匹配第一个类型
        
        return importance_info
    
    def adjust_threshold_for_important(self, threshold: float, clip: Dict) -> float:
        """为重要话题调整阈值"""
        importance = self.identify_importance(clip)
        if importance["is_important"]:
            adjusted = threshold + importance["threshold_adjustment"]
            logger.info(f"重要话题调整阈值: {threshold:.3f} -> {adjusted:.3f} ({importance['importance_type']})")
            return adjusted
        return threshold
```

```python
import time

class DebugEnhancement:

    def __init__(self):

            log_msg += f", 详情={details}"
        logger.debug(log_msg)
    
    def log_statistics(self, stats: Dict):
        """记录统计信息"""
        logger.info("评分统计:")
        for key, value in stats.items():
            logger.info(f"  {key}: {value}")
```
